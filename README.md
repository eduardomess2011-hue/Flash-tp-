local function triggerAlignCamera()
    local Workspace = game:GetService("Workspace")
    local character = player.Character
    if not character or not character:FindFirstChild("HumanoidRootPart") then 
        return 
    end
    
    local rootPart = character.HumanoidRootPart
    local playerPos = rootPart.Position
    
    local plotsFolder = Workspace:FindFirstChild("Plots")
    if not plotsFolder then return end

    local nearestPart = nil
    local shortestDistance = math.huge

    for _, plot in ipairs(plotsFolder:GetChildren()) do
        local animalPodiums = plot:FindFirstChild("AnimalPodiums")
        if animalPodiums then
            for _, child in ipairs(animalPodiums:GetDescendants()) do
                if child:IsA("BasePart") then
                    local colorName = tostring(child.BrickColor):lower()
                    if string.find(colorName, "green") or string.find(colorName, "lime") then
                        local distance = (playerPos - child.Position).Magnitude
                        if distance < shortestDistance then
                            shortestDistance = distance
                            nearestPart = child
                        end
                    end
                end
            end
        end
    end

    if nearestPart then
        local _, podiumYaw, _ = nearestPart.CFrame:ToOrientation()
        local targetPosition = nearestPart.Position + Vector3.new(0, 3, 0)
        local characterYaw = podiumYaw + math.rad(180)
        
        pcall(function()
            rootPart.CFrame = CFrame.new(targetPosition) * CFrame.Angles(0, characterYaw, 0)
        end)
        
        local humanoid = character:FindFirstChildOfClass("Humanoid")
        local backpack = player:FindFirstChild("Backpack")
        
        if backpack and humanoid then
            local carpetTool = backpack:FindFirstChild("carpettool") or backpack:FindFirstChild("Carpettool")
            if not carpetTool then
                for _, item in ipairs(backpack:GetChildren()) do
                    if item:IsA("Tool") and string.find(string.lower(item.Name), "carpet") then
                        carpetTool = item
                        break
                    end
                end
            end
            if carpetTool then
                pcall(function() humanoid:EquipTool(carpetTool) end)
            end
        end
        
        local camera = Workspace.CurrentCamera
        if camera then
            task.wait(0.1) 
            local pitchX = 20
            local rollZ = 0
            local radX = math.rad(pitchX)
            local radY = podiumYaw + math.rad(180)
            local radZ = math.rad(rollZ)
            
            task.spawn(function()
                for i = 1, 15 do
                    pcall(function()
                        camera.CameraType = Enum.CameraType.Custom
                        local currentPosition = camera.CFrame.Position
                        camera.CFrame = CFrame.new(currentPosition) * CFrame.fromOrientation(radX, radY, radZ)
                    end)
                    RunService.RenderStepped:Wait()
                end
            end)
        end
    end
end

local lastTick = tick()
local frames = 0

RunService.RenderStepped:Connect(function()
    frames = frames + 1
    local currentTick = tick()
    
    if currentTick - lastTick >= 1 then
        local fps = frames
        local ping = 0
    
        pcall(function()
            ping = math.round(Stats.Network.ServerStatsItem["Data Ping"]:GetValue())
        end)
        
        StatsLabel.Text = string.format("FPS: %d | PING: %dms", fps, ping)
        frames = 0
        lastTick = currentTick
    end
end)

local dragging, dragInput, dragStart, startPos

local function update(input)
    local delta = input.Position - dragStart
    MenuFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end

TopBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = MenuFrame.Position
        
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

MenuFrame.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
        dragInput = input
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        update(input)
    end
end)

-- ЛОГИКА ТАЙМИНГА И АВТОМАТИЧЕСКОГО СРАБАТЫВАНИЯ С КЛИКАМИ
-- =======================================================================
local activeTriggers = {}

ProximityPromptService.PromptButtonHoldBegan:Connect(function(prompt, who)
    if who ~= player or (not config.enabled and not config.giantEnabled) then return end
    if prompt.ActionText ~= "Steal" and prompt.Name ~= "Steal" then return end
    if activeTriggers[prompt] then return end
    
    activeTriggers[prompt] = true
    local startClock = os.clock()
    local ping = 0
    
    pcall(function()
        local rawPing = player:GetNetworkPing()
        if rawPing and rawPing > 0 then ping = rawPing end
    end)
    
    local pingCompensation = math.clamp(ping * 0.85, 0, 0.25)
    local targetProgressTime = prompt.HoldDuration * (config.timingPercent / 100)
    local exactFireTime = startClock + targetProgressTime - pingCompensation

    local fired = false
    local conn
    
    conn = RunService.Heartbeat:Connect(function()
        if not prompt or not prompt.Parent then
            conn:Disconnect()
            activeTriggers[prompt] = nil
            return
        end
        
        if os.clock() >= exactFireTime and not fired then
            fired = true
            conn:Disconnect()
            activeTriggers[prompt] = nil
            
            local char = player.Character
            local backpack = player:FindFirstChild("Backpack")
            if not char or not backpack then return end
            
            local hum = char:FindFirstChildOfClass("Humanoid")
            if not hum or hum.Health <= 0 then return end

            task.spawn(function()
                local flashTool = config.enabled and (char:FindFirstChild(config.toolName) or backpack:FindFirstChild(config.toolName))
                local giantTool = config.giantEnabled and (char:FindFirstChild(config.giantToolName) or backpack:FindFirstChild(config.giantToolName))

                hum:UnequipTools()
                task.wait(0.01)
                
                if flashTool then
                    flashTool.Parent = char
                    hum:EquipTool(flashTool)
                    task.wait(0.01)
                    for i = 1, 3 do 
                        pcall(function() flashTool:Activate() end) 
                        RunService.Heartbeat:Wait() 
                    end
                end
                
                if giantTool then
                    giantTool.Parent = char
                    hum:EquipTool(giantTool)
                    task.wait(0.01)
                    for i = 1, 3 do 
                        pcall(function() giantTool:Activate() end) 
                        RunService.Heartbeat:Wait() 
                    end
                end

                task.wait(0.05)
                if flashTool then pcall(function() flashTool.Parent = backpack end) end
                if giantTool then pcall(function() giantTool.Parent = backpack end) end
            end)
        end
    end)

    prompt.PromptButtonHoldEnded:Connect(function()
        if not fired then
            if conn then conn:Disconnect() end
            activeTriggers[prompt] = nil
        end
    end)
end)
